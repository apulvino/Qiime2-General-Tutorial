#!/bin/bash

  #SBATCH --account=b1057
  #SBATCH --partition=b1057
  #SBATCH --job-name=qiime_summaryinfo
  #SBATCH --output=qiime_summaryinfo.out
  #SBATCH --error=qiime_summaryinfo.err
  #SBATCH --time=00:50:00
  #SBATCH --mem=10G
  #SBATCH --ntasks-per-node=1
  #SBATCH --cpus-per-task=2
  #SBATCH --mail-user=firstname.lastname@northwestern.edu
  #SBATCH --mail-type=END

  module load qiime2/2022.8
  ## See man qiime diversity alpha-phylogenetic for options to flag: --p-metric
  qiime diversity alpha-phylogenetic --i-table /path/to/your/output/table_dada2_nomito_nochloro.qza \
  --i-phylogeny /path/to/your/output/rooted_tree.qza \
  --o-alpha-diversity /path/to/your/output/pd_diversity --p-metric faith_pd

  
  qiime tools export --input-path /path/to/your/output/pd_diversity.qza --output-path /path/to/your/output/pd_diversity

  qiime diversity alpha --i-table /path/to/your/output/table_dada2_nomito_nochloro.qza \
  --o-alpha-diversity /path/to/your/output/shannon --p-metric shannon

  qiime tools export --input-path /path/to/your/output/shannon.qza --output-path /path/to/your/output/shannon
